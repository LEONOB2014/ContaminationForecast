[docs]    def getContaminantes(self,conn):
        cur = conn.cursor();
        cur.execute(""" SELECT table_name FROM information_schema.tables WHERE table_schema='public' AND table_type='BASE TABLE'; """);
        rows = cur.fetchall();
        cur.close();

        return rows
