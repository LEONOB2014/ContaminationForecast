        "Obtains the date format"
        firstData = 11
        f = open(fileName)
        values = f.readlines()[11:]
        for line in values:
            allDate = (line.rstrip().split(',')[0]).split('/')
            #print allDate
            if int(allDate[0]) > 12:
                return "DD/MM/YYY/HH24"
                break

                if int(allDate[1]) > 12:
                    return "MM/DD/YYY/HH24"
